#### Discussion:
The first idea I tried is the Brute Force, though I knew it wouldn't be the best method. Maybe the code is correct cuz no wrong answer appeared before Time Limit Exceeded.<br>
Then I read some explation about the Palindrome and then knew how to find it. The key point is the middle item of the string. Keep the middle item or +(-)1 can get two Palindrome String and then get the closest one.
Some difficulties I met with in coding:<br>
* First is data format transformation. I forget to use `string` as dataset directly, then I tried `stringstream`, which made me written to much unnecessary code. Guilty!
* Second is `memcpy()`, I tried to use this function to make a copy. However, when I checked the memory address, some items was lost. So I used a loop to make copy.

class Solution {
public:
    string nearestPalindromic(string n) {
        char n_arr[19];
        stringstream stream; 
        stream << n;
        stream >> n_arr;
        char n_arr_copy[19];
        // memcpy(n_arr_copy,n_arr,sizeof(n_arr));
        for(int i=0;i<19;i++){
            n_arr_copy[i] = n_arr[i];
        }
        int len = strlen(n_arr);
        for (int i = 0; i < len / 2; i++) {
            n_arr_copy[len - 1 - i] = n_arr_copy[i];
        }
        stream.clear();
        stream.str("");
        stream << n_arr_copy;
        long long standard;
        stream >> standard;
        stream.clear();
        stream.str("");
        long long n_long;
        stream << n;
        stream >> n_long;
        stream.clear();
        stream.str("");
        long long high;
        long long low;
        if(standard > n_long){
            low = findLower(n_arr);
            high = standard;
        }else if (standard < n_long){
            high = findHigher(n_arr);
            low = standard;
        }else{
            high = findHigher(n_arr);
            low = findLower(n_arr);
        }
        stream << high;
        string shigh;
        stream >> shigh;
        stream.clear();
        stream.str("");
        stream << low;
        string slow;
        stream >> slow;
        return abs(high-n_long) >= abs(low-n_long) ? slow : shigh;
    }
    
    long long findHigher(char* n_arr){
        char n_arr_copy[19];
        // memcpy(n_arr_copy,n_arr,sizeof(n_arr));
        for(int i=0;i<19;i++){
            n_arr_copy[i] = n_arr[i];
        }
        int len = strlen(n_arr);
        for (int j = (len - 1) / 2; j >= 0; j--) {
            if (++n_arr_copy[j] > '9') {
                n_arr_copy[j] = '0';
            } else {
                break;
            }
        }
        for (int k = 0; k < len / 2; k++) {
            n_arr_copy[len - 1 - k] = n_arr_copy[k];
        }
        
        char n_result[19];
        if (n_arr_copy[0] == '0') {
            for(int i=0;i<len+1;i++){
                if(i==0){
                   n_result[i] = '1'; 
                }else{
                    n_result[i] = '0';
                }
            }
            n_result[len+1] = '\0';
            for (int k = 0; k < (len+1) / 2; k++) {
                n_result[len - k] = n_result[k];
            }
            stringstream stream;
            stream << n_result;
            long long result;
            stream >> result;
            return result;
        }else{
            stringstream stream;
            stream << n_arr_copy;
            long long result;
            stream >> result;
            return result;
        }
    }
    
    long long findLower(char* n_arr){
        char n_arr_copy[19];
        // memcpy(n_arr_copy,n_arr,sizeof(n_arr));
        for(int i=0;i<19;i++){
            n_arr_copy[i] = n_arr[i];
        }
        int len = strlen(n_arr);
        for (int j = (len - 1) / 2; j >= 0; j--) {
            if (--n_arr_copy[j] < '0') {
                n_arr_copy[j] = '9';
            } else {
                break;
            }
        }
        for (int k = 0; k < len / 2; k++) {
            n_arr_copy[len - 1 - k] = n_arr_copy[k];
        }

        if(len==1){
            stringstream stream;
            stream << n_arr_copy;
            long long result;
            stream >> result;
            return result;
        }else{
            char n_result[19];
            if (n_arr_copy[0] == '0') {
                for(int i=0;i<len-1;i++){
                    n_result[i] = '9';
                }
                n_result[len-1] = '\0';
                stringstream stream;
                stream << n_result;
                long long result;
                stream >> result;
                return result;
            }else{
                stringstream stream;
                stream << n_arr_copy;
                long long result;

                return result;
            }
        }
